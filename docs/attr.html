"Cахар", для работы с атрибутами.
ctx.a = 2; ctx.b = 1; ctx.color = '#000'; ctx.$loop(function (){ ctx.b = !ctx.b; ctx.color = ctx.b ? '#000' : '#f60'; }, 1000);
  • @style = 'color: '+ctx.color; @attrName = 'newVal'; — полностью заменяет значение атрибута
  • @class += ' bar'; @attr += 'substring'; — конкатенация
  • @prop = value; — установить свойство объекта, например checked: @checked = (ctx.a + ctx.b == 3);
